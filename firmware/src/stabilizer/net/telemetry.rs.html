1//! Stabilizer Telemetry Capabilities
2//!
3//! # Design
4//! Telemetry is reported regularly using an MQTT client. All telemetry is reported in SI units
5//! using standard JSON format.
6//!
7//! In order to report ADC/DAC codes generated during the DSP routines, a telemetry buffer is
8//! employed to track the latest codes. Converting these codes to SI units would result in
9//! repetitive and unnecessary calculations within the DSP routine, slowing it down and limiting
10//! sampling frequency. Instead, the raw codes are stored and the telemetry is generated as
11//! required immediately before transmission. This ensures that any slower computation required
12//! for unit conversion can be off-loaded to lower priority tasks.
13use crate::hardware::metadata::ApplicationMetadata;
14use heapless::String;
15use minimq::Publication;
16use serde::Serialize;
17
18use super::NetworkReference;
19use crate::hardware::{adc::AdcCode, afe::Gain, dac::DacCode, SystemTimer};
20
21/// Default metadata message if formatting errors occur.
22const DEFAULT_METADATA: &str = "{\"message\":\"Truncated: See USB terminal\"}";
23
24/// The telemetry client for reporting telemetry data over MQTT.
25pub struct TelemetryClient {
26    mqtt: minimq::Minimq<
27        'static,
28        NetworkReference,
29        SystemTimer,
30        minimq::broker::NamedBroker<NetworkReference>,
31    >,
32    prefix: &'static str,
33    meta_published: bool,
34    metadata: &'static ApplicationMetadata,
35}
36
37/// The telemetry buffer is used for storing sample values during execution.
38///
39/// # Note
40/// These values can be converted to SI units immediately before reporting to save processing time.
41/// This allows for the DSP process to continually update the values without incurring significant
42/// run-time overhead during conversion to SI units.
43#[derive(Clone, Default)]
44pub struct TelemetryBuffer {
45    /// The latest input sample on ADC0/ADC1.
46    pub adcs: [AdcCode; 2],
47    /// The latest output code on DAC0/DAC1.
48    pub dacs: [DacCode; 2],
49    /// The latest digital input states during processing.
50    pub digital_inputs: [bool; 2],
51}
52
53/// The telemetry structure is data that is ultimately reported as telemetry over MQTT.
54///
55/// # Note
56/// This structure should be generated on-demand by the buffer when required to minimize conversion
57/// overhead.
58#[derive(Serialize)]
59pub struct Telemetry {
60    /// Most recent input voltage measurement.
61    pub adcs: [f32; 2],
62
63    /// Most recent output voltage.
64    pub dacs: [f32; 2],
65
66    /// Most recent digital input assertion state.
67    pub digital_inputs: [bool; 2],
68
69    /// The CPU temperature in degrees Celsius.
70    pub cpu_temp: f32,
71}
72
73impl TelemetryBuffer {
74    /// Convert the telemetry buffer to finalized, SI-unit telemetry for reporting.
75    ///
76    /// # Args
77    /// * `afe0` - The current AFE configuration for channel 0.
78    /// * `afe1` - The current AFE configuration for channel 1.
79    /// * `cpu_temp` - The current CPU temperature.
80    ///
81    /// # Returns
82    /// The finalized telemetry structure that can be serialized and reported.
83    pub fn finalize(self, afe0: Gain, afe1: Gain, cpu_temp: f32) -> Telemetry {
84        let in0_volts = f32::from(self.adcs[0]) / afe0.gain();
85        let in1_volts = f32::from(self.adcs[1]) / afe1.gain();
86
87        Telemetry {
88            cpu_temp,
89            adcs: [in0_volts, in1_volts],
90            dacs: [self.dacs[0].into(), self.dacs[1].into()],
91            digital_inputs: self.digital_inputs,
92        }
93    }
94}
95
96impl TelemetryClient {
97    /// Construct a new telemetry client.
98    ///
99    /// # Args
100    /// * `mqtt` - The MQTT client
101    /// * `prefix` - The device prefix to use for MQTT telemetry reporting.
102    ///
103    /// # Returns
104    /// A new telemetry client.
105    pub fn new(
106        mqtt: minimq::Minimq<
107            'static,
108            NetworkReference,
109            SystemTimer,
110            minimq::broker::NamedBroker<NetworkReference>,
111        >,
112        prefix: &'static str,
113        metadata: &'static ApplicationMetadata,
114    ) -> Self {
115        Self {
116            mqtt,
117            meta_published: false,
118            prefix,
119            metadata,
120        }
121    }
122
123    /// Publish telemetry over MQTT
124    ///
125    /// # Note
126    /// Telemetry is reported in a "best-effort" fashion. Failure to transmit telemetry will cause
127    /// it to be silently dropped.
128    ///
129    /// # Args
130    /// * `telemetry` - The telemetry to report
131    pub fn publish<T: Serialize>(&mut self, telemetry: &T) {
132        let mut topic: String<128> = self.prefix.try_into().unwrap();
133        topic.push_str("/telemetry").unwrap();
134
135        self.mqtt
136            .client()
137            .publish(minimq::Publication::new(&topic, |buf: &mut [u8]| {
138                serde_json_core::to_slice(telemetry, buf)
139            }))
140            .map_err(|e| log::error!("Telemetry publishing error: {:?}", e))
141            .ok();
142    }
143
144    /// Update the telemetry client
145    ///
146    /// # Note
147    /// This function is provided to force the underlying MQTT state machine to process incoming
148    /// and outgoing messages. Without this, the client will never connect to the broker. This
149    /// should be called regularly.
150    pub fn update(&mut self) {
151        match self.mqtt.poll(|_client, _topic, _message, _properties| {}) {
152            Err(minimq::Error::Network(
153                smoltcp_nal::NetworkError::TcpConnectionFailure(
154                    smoltcp_nal::smoltcp::socket::tcp::ConnectError::Unaddressable
155                ),
156            )) => {}
157
158            Err(error) => log::info!("Unexpected error: {:?}", error),
159            _ => {}
160        }
161
162        if !self.mqtt.client().is_connected() {
163            self.meta_published = false;
164            return;
165        }
166
167        // Publish application metadata
168        if !self.meta_published
169            && self.mqtt.client().can_publish(minimq::QoS::AtMostOnce)
170        {
171            let Self {
172                ref mut mqtt,
173                metadata,
174                ..
175            } = self;
176
177            let mut topic: String<128> = self.prefix.try_into().unwrap();
178            topic.push_str("/meta").unwrap();
179
180            if mqtt
181                .client()
182                .publish(Publication::new(&topic, |buf: &mut [u8]| {
183                    serde_json_core::to_slice(&metadata, buf)
184                }))
185                .is_err()
186            {
187                // Note(unwrap): We can guarantee that this message will be sent because we checked
188                // for ability to publish above.
189                mqtt.client()
190                    .publish(Publication::new(
191                        &topic,
192                        DEFAULT_METADATA.as_bytes(),
193                    ))
194                    .unwrap();
195            }
196
197            self.meta_published = true;
198        }
199    }
200}
