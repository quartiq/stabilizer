116/// Construct the global metadata.
117///
118/// # Note
119/// This may only be called once.
120///
121/// # Args
122/// * `hardware_version` - The hardware version detected.
123///
124/// # Returns
125/// A reference to the global metadata.
126pub fn metadata(version: &'static str) -> &'static ApplicationMetadata {
127    cortex_m::singleton!(: ApplicationMetadata = ApplicationMetadata {
128        firmware_version: build_info::GIT_VERSION.unwrap_or("Unspecified"),
129        rust_version: build_info::RUSTC_VERSION,
130        profile: build_info::PROFILE,
131        git_dirty: build_info::GIT_DIRTY.unwrap_or(false),
132        features: build_info::FEATURES_STR,
133        hardware_version: version,
134        panic_info: panic_persist::get_panic_message_utf8().unwrap_or("None"),
135    })
136    .unwrap()
137}

139#[derive(strum::IntoStaticStr)]
140pub enum HardwareVersion {
141    Rev1_0,
142    Rev1_1,
143    Rev1_2,
144    Rev1_3,
145    Unknown(u8),
146}
147
148impl From<&[bool]> for HardwareVersion {
149    fn from(bits: &[bool]) -> Self {
150        match bits.iter().rev().fold(0, |v, b| (v << 1) | *b as u8) {
151            0b000 => HardwareVersion::Rev1_0,
152            0b001 => HardwareVersion::Rev1_1,
153            0b010 => HardwareVersion::Rev1_2,
154            0b011 => HardwareVersion::Rev1_3,
155            other => HardwareVersion::Unknown(other),
156        }
157    }
158}
159
160impl core::fmt::Display for HardwareVersion {
161    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
162        match self {
163            HardwareVersion::Rev1_0 => write!(f, "v1.0"),
164            HardwareVersion::Rev1_1 => write!(f, "v1.1"),
165            HardwareVersion::Rev1_2 => write!(f, "v1.2"),
166            HardwareVersion::Rev1_3 => write!(f, "v1.3"),
167            HardwareVersion::Unknown(other) => {
168                write!(f, "Unknown ({:#b})", other)
169            }
170        }
171    }
172}

174#[inline(never)]
175#[panic_handler]
176fn panic(info: &core::panic::PanicInfo) -> ! {
177    use core::{
178        fmt::Write,
179        sync::atomic::{AtomicBool, Ordering},
180    };
181    use cortex_m::asm;
182    use rtt_target::{ChannelMode, UpChannel};
183
184    cortex_m::interrupt::disable();
185
186    // Recursion protection
187    static PANICKED: AtomicBool = AtomicBool::new(false);
188    while PANICKED.load(Ordering::Relaxed) {
189        asm::bkpt();
190    }
191    PANICKED.store(true, Ordering::Relaxed);
192
193    // Turn on both red LEDs, FP_LED_1, FP_LED_3
194    let gpiod = unsafe { &*hal::stm32::GPIOD::ptr() };
195    gpiod.odr.modify(|_, w| w.odr6().high().odr12().high());
196
197    // Analogous to panic-rtt-target
198    if let Some(mut channel) = unsafe { UpChannel::conjure(0) } {
199        channel.set_mode(ChannelMode::BlockIfFull);
200        writeln!(channel, "{info}").ok();
201    }
202
203    panic_persist::report_panic_info(info);
204
205    // Abort
206    asm::udf();
207    // Halt
208    // loop { core::sync::atomic::compiler_fence(Ordering::SeqCst); }
209}
