1use super::hal;
2/// Shared Internal ADC Support
3///
4/// # Description
5/// This module provides an abstraction to share ownership of a single ADC peripheral with multiple
6/// ADC channels attached to it.
7///
8/// The design of this module mimics that of [`shared-bus`].
9///
10/// First, the shared ADC is created with the use of a macro, which places the ADC peripheral into
11/// a mutable, static (singleton) location. Then, individual channels are created by passing in the
12/// associated ADC input pin to the [SharedAdc::create_channel()] function to generate an
13/// [AdcChannel]. The [AdcChannel]'s ownership can then be moved to any required drivers.
14///
15/// ## Synchronization
16/// If the multiple priorities utilize the ADC that results in resource pre-emption, pre-emption is
17/// protected against through the use of an atomic bool. Attempting to utilize the ADC from a
18/// higher priority level while it is in use at a lower level will result in a [AdcError::InUse].
19use embedded_hal_02::adc::{Channel, OneShot};
20
21#[derive(Debug, Copy, Clone)]
22pub enum AdcError {
23    /// Indicates that the ADC is already in use
24    InUse,
25}
26
27/// A single channel on an ADC peripheral.
28pub struct AdcChannel<'a, Adc, PIN> {
29    pin: PIN,
30    slope: f32,
31    mutex: &'a spin::Mutex<hal::adc::Adc<Adc, hal::adc::Enabled>>,
32}
33
34impl<Adc, PIN> AdcChannel<'_, Adc, PIN>
35where
36    PIN: Channel<Adc, ID = u8>,
37    hal::adc::Adc<Adc, hal::adc::Enabled>: OneShot<Adc, u32, PIN>,
38    <hal::adc::Adc<Adc, hal::adc::Enabled> as OneShot<Adc, u32, PIN>>::Error:
39        core::fmt::Debug,
40{
41    /// Read the ADC channel and normalize the result.
42    ///
43    /// # Returns
44    /// The normalized ADC measurement as a ratio of full-scale.
45    pub fn read_normalized(&mut self) -> Result<f32, AdcError> {
46        self.read_raw().map(|code| code as f32 / self.slope)
47    }
48
49    /// Read the raw ADC sample for the channel.
50    ///
51    /// # Returns
52    /// The raw ADC code measured on the channel.
53    pub fn read_raw(&mut self) -> Result<u32, AdcError> {
54        let mut adc = self.mutex.try_lock().ok_or(AdcError::InUse)?;
55        Ok(adc.read(&mut self.pin).unwrap())
56    }
57}
58
59/// An ADC peripheral that can provide ownership of individual channels for sharing between
60/// drivers.
61pub struct SharedAdc<Adc> {
62    mutex: spin::Mutex<hal::adc::Adc<Adc, hal::adc::Enabled>>,
63    slope: f32,
64}
65
66impl<Adc> SharedAdc<Adc> {
67    /// Construct a new shared ADC driver.
68    ///
69    /// # Args
70    /// * `slope` - The slope of the ADC conversion transfer function.
71    /// * `adc` - The ADC peripheral to share.
72    pub fn new(slope: f32, adc: hal::adc::Adc<Adc, hal::adc::Enabled>) -> Self {
73        Self {
74            slope,
75            mutex: spin::Mutex::new(adc),
76        }
77    }
78
79    /// Allocate an ADC channel for usage.
80    ///
81    /// # Args
82    /// * `pin` - The ADC input associated with the desired ADC channel. Often, this is a GPIO pin.
83    ///
84    /// # Returns
85    /// An instantiated [AdcChannel] whose ownership can be transferred to other drivers.
86    pub fn create_channel<PIN: Channel<Adc, ID = u8>>(
87        &self,
88        pin: PIN,
89    ) -> AdcChannel<'_, Adc, PIN> {
90        AdcChannel {
91            pin,
92            slope: self.slope,
93            mutex: &self.mutex,
94        }
95    }
96}
