1use embedded_hal_02::blocking::{delay::DelayUs, i2c::WriteRead};
2
3// The EEPROM is a variant without address bits, so the 3 LSB of this word are "dont-cares".
4const I2C_ADDR: u8 = 0x50;
5
6// The MAC address is stored in the last 6 bytes of the 256 byte address space.
7const MAC_POINTER: u8 = 0xFA;
8
9pub fn read_eui48<T>(i2c: &mut T, delay: &mut impl DelayUs<u32>) -> [u8; 6]
10where
11    T: WriteRead,
12{
13    let mut previous_read: Option<[u8; 6]> = None;
14    // On Stabilizer v1.1 and earlier hardware, there is a fault where the I2C bus is not connected
15    // to the CPU until the P12V0A rail enables, which can take many seconds, or may never come up
16    // at all. During these transient turn-on conditions, we may fail the I2C read operation. To
17    // accomodate this, we repeat the I2C read for a set number of attempts with a fixed delay
18    // between them. Then, we wait for the bus to stabilize by waiting until the MAC address
19    // read-out is identical for two consecutive reads.
20    for _ in 0..40 {
21        let mut buffer = [0u8; 6];
22        if i2c
23            .write_read(I2C_ADDR, &[MAC_POINTER], &mut buffer)
24            .is_ok()
25        {
26            if let Some(old_read) = previous_read
27                && old_read == buffer
28            {
29                return buffer;
30            }
31
32            previous_read.replace(buffer);
33        } else {
34            // Remove any pending previous read if we failed the last attempt.
35            previous_read.take();
36        }
37
38        delay.delay_us(100_000);
39    }
40
41    panic!("Failed to read MAC address");
42}
