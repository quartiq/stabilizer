1//! The sampling timer is used for managing ADC sampling and external reference timestamping.
2use super::hal;
3use num_enum::TryFromPrimitive;

5use hal::stm32::{
6    // TIM1 and TIM8 have identical registers.
7    tim1 as __tim8,

14/// The event that should generate an external trigger from the peripheral.
15#[allow(dead_code)]
16pub enum TriggerGenerator {
17    Reset = 0b000,
18    Enable = 0b001,
19    Update = 0b010,
20    ComparePulse = 0b011,
21    Ch1Compare = 0b100,
22    Ch2Compare = 0b101,
23    Ch3Compare = 0b110,
24    Ch4Compare = 0b111,
25}
26
27/// Selects the trigger source for the timer peripheral.
28#[allow(dead_code)]
29pub enum TriggerSource {
30    Trigger0 = 0,
31    Trigger1 = 0b01,
32    Trigger2 = 0b10,
33    Trigger3 = 0b11,
34}
35
36/// Prescalers for externally-supplied reference clocks.
37#[allow(dead_code)]
38#[derive(TryFromPrimitive)]
39#[repr(u8)]
40pub enum Prescaler {
41    Div1 = 0b00,
42    Div2 = 0b01,
43    Div4 = 0b10,
44    Div8 = 0b11,
45}
46
47/// Optional slave operation modes of a timer.
48#[allow(dead_code)]
49pub enum SlaveMode {
50    Disabled = 0,
51    Trigger = 0b0110,
52}
53
54/// Optional input capture preconditioning filter configurations.
55#[allow(dead_code)]
56pub enum InputFilter {
57    Div1N1 = 0b0000,
58    Div1N8 = 0b0011,
59}

61macro_rules! timer_channels {
62    ($name:ident, $TY:ident, $size:ty) => {
63        paste::paste! {
64
65            /// The timer used for managing ADC sampling.
66            pub struct $name {
67                timer: hal::timer::Timer<hal::stm32::[< $TY >]>,
68                channels: Option<[< $TY:lower >]::Channels>,
69                update_event: Option<[< $TY:lower >]::UpdateEvent>,
70            }
71
72            impl $name {
73                /// Construct the sampling timer.
74                #[allow(dead_code)]
75                pub fn new(mut timer: hal::timer::Timer<hal::stm32::[< $TY>]>) -> Self {
76                    timer.pause();
77
78                    Self {
79                        timer,
80                        // Note(unsafe): Once these channels are taken, we guarantee that we do not
81                        // modify any of the underlying timer channel registers, as ownership of the
82                        // channels is now provided through the associated channel structures. We
83                        // additionally guarantee this can only be called once because there is only
84                        // one Timer2 and this resource takes ownership of it once instantiated.
85                        channels: unsafe { Some([< $TY:lower >]::Channels::new()) },
86                        update_event: unsafe { Some([< $TY:lower >]::UpdateEvent::new()) },
87                    }
88                }
89
90                /// Get the timer capture/compare channels.
91                #[allow(dead_code)]
92                pub fn channels(&mut self) -> [< $TY:lower >]::Channels {
93                    self.channels.take().unwrap()
94                }
95
96                /// Get the timer update event.
97                #[allow(dead_code)]
98                pub fn update_event(&mut self) -> [< $TY:lower >]::UpdateEvent {
99                    self.update_event.take().unwrap()
100                }
101
102                /// Get the period of the timer.
103                #[allow(dead_code)]
104                pub fn get_period(&self) -> $size {
105                    let regs = unsafe { &*hal::stm32::$TY::ptr() };
106                    regs.arr.read().arr().bits()
107                }
108
109                /// Manually set the period of the timer.
110                #[allow(dead_code)]
111                pub fn set_period_ticks(&mut self, period: $size) {
112                    let regs = unsafe { &*hal::stm32::$TY::ptr() };
113                    regs.arr.write(|w| w.arr().bits(period));
114
115                    // Force the new period to take effect immediately.
116                    self.timer.apply_freq();
117                }
118
119                /// Clock the timer from an external source.
120                ///
121                /// # Note:
122                /// * Currently, only an external source applied to ETR is supported.
123                ///
124                /// # Args
125                /// * `prescaler` - The prescaler to use for the external source.
126                #[allow(dead_code)]
127                pub fn set_external_clock(&mut self, prescaler: Prescaler) {
128                    let regs = unsafe { &*hal::stm32::$TY::ptr() };
129                    regs.smcr.modify(|_, w| w.etps().bits(prescaler as u8).ece().set_bit());
130
131                    // Clear any other prescaler configuration.
132                    regs.psc.write(|w| w.psc().bits(0));
133                }
134
135                /// Start the timer.
136                #[allow(dead_code)]
137                pub fn start(&mut self) {
138                    // Force a refresh of the frequency settings.
139                    self.timer.apply_freq();
140                    self.timer.reset_counter();
141
142                    self.timer.resume();
143                }
144
145                /// Configure the timer peripheral to generate a trigger based on the provided
146                /// source.
147                #[allow(dead_code)]
148                pub fn generate_trigger(&mut self, source: TriggerGenerator) {
149                    let regs = unsafe { &*hal::stm32::$TY::ptr() };
150                    // Note(unsafe) The TriggerGenerator enumeration is specified such that this is
151                    // always in range.
152                    regs.cr2.modify(|_, w| w.mms().bits(source as u8));
153
154                }
155
156                /// Select a trigger source for the timer peripheral.
157                #[allow(dead_code)]
158                pub fn set_trigger_source(&mut self, source: TriggerSource) {
159                    let regs = unsafe { &*hal::stm32::$TY::ptr() };
160                    // Note(unsafe) The TriggerSource enumeration is specified such that this is
161                    // always in range.
162                    regs.smcr.modify(|_, w| unsafe { w.ts().bits(source as u8) } );
163                }
164
165                #[allow(dead_code)]
166                pub fn set_slave_mode(&mut self, source: TriggerSource, mode: SlaveMode) {
167                    let regs = unsafe { &*hal::stm32::$TY::ptr() };
168                    // Note(unsafe) The TriggerSource and SlaveMode enumerations are specified such
169                    // that they are always in range.
170                    regs.smcr.modify(|_, w| unsafe { w.sms().bits(mode as u8).ts().bits(source as u8) } );
171                }
172            }
173
174            pub mod [< $TY:lower >] {
175                use super::hal::{stm32::$TY, dma::{traits::TargetAddress, PeripheralToMemory, dma::DMAReq}};
176
177                pub struct UpdateEvent {}
178
179                impl UpdateEvent {
180                    /// Create a new update event
181                    ///
182                    /// # Safety
183                    /// This is only safe to call once.
184                    #[allow(dead_code)]
185                    pub unsafe fn new() -> Self {
186                        Self {}
187                    }
188
189                    /// Enable DMA requests upon timer updates.
190                    #[allow(dead_code)]
191                    pub fn listen_dma(&self) {
192                        // Note(unsafe): We perform only atomic operations on the timer registers.
193                        let regs = unsafe { &*<$TY>::ptr() };
194                        regs.dier.modify(|_, w| w.ude().set_bit());
195                    }
196
197                    /// Trigger a DMA request manually
198                    #[allow(dead_code)]
199                    pub fn trigger(&self) {
200                        let regs = unsafe { &*<$TY>::ptr() };
201                        regs.egr.write(|w| w.ug().set_bit());
202                    }
203                }
204
205                /// The channels representing the timer.
206                pub struct Channels {
207                    pub ch1: Channel1,
208                    pub ch2: Channel2,
209                    pub ch3: Channel3,
210                    pub ch4: Channel4,
211                }
212
213                impl Channels {
214                    /// Construct a new set of channels.
215                    ///
216                    /// # Safety
217                    /// This is only safe to call once.
218                    #[allow(dead_code)]
219                    pub unsafe fn new() -> Self {
220                        unsafe { Self {
221                            ch1: Channel1::new(),
222                            ch2: Channel2::new(),
223                            ch3: Channel3::new(),
224                            ch4: Channel4::new(),
225                        } }
226                    }
227                }
228
229                timer_channels!(1, $TY, ccmr1, $size);
230                timer_channels!(2, $TY, ccmr1, $size);
231                timer_channels!(3, $TY, ccmr2, $size);
232                timer_channels!(4, $TY, ccmr2, $size);
233            }
234        }
235    };

237    ($index:expr, $TY:ty, $ccmrx:expr, $size:ty) => {
238        paste::paste! {
239            pub use super::[< __ $TY:lower >]::[< $ccmrx _input >]::[< CC $index S_A>] as [< CaptureSource $index >];
240
241            /// A capture/compare channel of the timer.
242            pub struct [< Channel $index >] {}
243
244            /// A capture channel of the timer.
245            pub struct [< Channel $index InputCapture>] {}
246
247            impl [< Channel $index >] {
248                /// Construct a new timer channel.
249                ///
250                /// Note(unsafe): This function must only be called once. Once constructed, the
251                /// constructee guarantees to never modify the timer channel.
252                #[allow(dead_code)]
253                unsafe fn new() -> Self {
254                    Self {}
255                }
256
257                /// Allow the channel to generate DMA requests.
258                #[allow(dead_code)]
259                pub fn listen_dma(&self) {
260                    let regs = unsafe { &*<$TY>::ptr() };
261                    regs.dier.modify(|_, w| w.[< cc $index de >]().set_bit());
262                }
263
264                /// Operate the channel as an output-compare.
265                ///
266                /// # Args
267                /// * `value` - The value to compare the sampling timer's counter against.
268                #[allow(dead_code)]
269                pub fn to_output_compare(&self, value: $size) {
270                    let regs = unsafe { &*<$TY>::ptr() };
271                    let arr = regs.arr.read().bits() as $size;
272                    assert!(value <= arr);
273                    regs.ccr[$index - 1].write(|w| w.ccr().bits(value));
274                    regs.[< $ccmrx _output >]()
275                        .modify(|_, w| unsafe { w.[< cc $index s >]().bits(0) });
276                }
277
278                /// Operate the channel in input-capture mode.
279                ///
280                /// # Args
281                /// * `input` - The input source for the input capture event.
282                #[allow(dead_code)]
283                pub fn into_input_capture(self, input: [< CaptureSource $index >]) -> [< Channel $index InputCapture >]{
284                    let regs = unsafe { &*<$TY>::ptr() };
285
286                    regs.[< $ccmrx _input >]().modify(|_, w| w.[< cc $index s>]().variant(input));
287
288                    [< Channel $index InputCapture >] {}
289                }
290            }
291
292            impl [< Channel $index InputCapture >] {
293                /// Get the latest capture from the channel.
294                #[allow(dead_code)]
295                pub fn latest_capture(&mut self) -> Result<Option<$size>, Option<$size>> {
296                    // Note(unsafe): This channel owns all access to the specific timer channel.
297                    // Only atomic operations on completed on the timer registers.
298                    let regs = unsafe { &*<$TY>::ptr() };
299
300                    if regs.sr.read().[< cc $index if >]().bit_is_set() {
301                        // Read the capture value. Reading the captured value clears the flag in the
302                        // status register automatically.
303                        let result = regs.ccr[$index - 1].read().ccr().bits();
304
305                        // Read SR again to check for a potential over-capture. Return an error in
306                        // that case.
307                        let sr = regs.sr.read();
308                        if sr.[< cc $index of >]().bit_is_set() {
309                            // NOTE(unsafe) write-back is safe
310                            regs.sr.write(|w| unsafe { w.bits(sr.bits()) }.[< cc $index of >]().clear_bit());
311                            Err(Some(result))
312                        } else {
313                            Ok(Some(result))
314                        }
315                    } else {
316                        Ok(None)
317                    }
318                }
319
320                /// Allow the channel to generate DMA requests.
321                #[allow(dead_code)]
322                pub fn listen_dma(&self) {
323                    // Note(unsafe): This channel owns all access to the specific timer channel.
324                    // Only atomic operations on completed on the timer registers.
325                    let regs = unsafe { &*<$TY>::ptr() };
326                    regs.dier.modify(|_, w| w.[< cc $index de >]().set_bit());
327                }
328
329                /// Enable the input capture to begin capturing timer values.
330                #[allow(dead_code)]
331                pub fn enable(&mut self) {
332                    // Read the latest input capture to clear any pending data in the register.
333                    let _ = self.latest_capture();
334
335                    // Note(unsafe): This channel owns all access to the specific timer channel.
336                    // Only atomic operations on completed on the timer registers.
337                    let regs = unsafe { &*<$TY>::ptr() };
338                    regs.ccer.modify(|_, w| w.[< cc $index e >]().set_bit());
339                }
340
341                /// Check if an over-capture event has occurred.
342                #[allow(dead_code)]
343                pub fn check_overcapture(&self) -> bool {
344                    // Note(unsafe): This channel owns all access to the specific timer channel.
345                    // Only atomic operations on completed on the timer registers.
346                    let regs = unsafe { &*<$TY>::ptr() };
347                    regs.sr.read().[< cc $index of >]().bit_is_set()
348                }
349
350                /// Configure the input capture input pre-filter.
351                ///
352                /// # Args
353                /// * `filter` - The desired input filter stage configuration. Defaults to disabled.
354                #[allow(dead_code)]
355                pub fn configure_filter(&mut self, filter: super::InputFilter) {
356                    // Note(unsafe): This channel owns all access to the specific timer channel.
357                    // Only atomic operations on completed on the timer registers.
358                    let regs = unsafe { &*<$TY>::ptr() };
359                    regs.[< $ccmrx _input >]().modify(|_, w| w.[< ic $index f >]().bits(filter as u8));
360                }
361
362                /// Configure the input capture prescaler.
363                ///
364                /// # Args
365                /// * `psc` - Prescaler exponent.
366                #[allow(dead_code)]
367                pub fn configure_prescaler(&mut self, prescaler: super::Prescaler) {
368                    // Note(unsafe): This channel owns all access to the specific timer channel.
369                    // Only atomic operations on completed on the timer registers.
370                    let regs = unsafe { &*<$TY>::ptr() };
371                    // Note(unsafe): Enum values are all valid.
372                    #[allow(unused_unsafe)]
373                    regs.[< $ccmrx _input >]().modify(|_, w| unsafe {
374                        w.[< ic $index psc >]().bits(prescaler as u8)});
375                }
376            }
377
378            // Note(unsafe): This manually implements DMA support for input-capture channels. This
379            // is safe as it is only completed once per channel and each DMA request is allocated to
380            // each channel as the owner.
381            unsafe impl TargetAddress<PeripheralToMemory> for [< Channel $index InputCapture >] {
382                type MemSize = $size;
383
384                const REQUEST_LINE: Option<u8> = Some(DMAReq::[< $TY:camel Ch $index >]as u8);
385
386                fn address(&self) -> usize {
387                    let regs = unsafe { &*<$TY>::ptr() };
388                    &regs.ccr[$index - 1] as *const _ as usize
389                }
390            }
391        }
392    };
