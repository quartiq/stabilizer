1//! The HRTimer (High Resolution Timer) is used to generate IO_Update pulses to the Pounder DDS.
2use super::hal::{
3    self,
4    rcc::{ResetEnable, rec},
5};
6
7/// A HRTimer output channel.
8#[allow(dead_code)]
9pub enum Channel {
10    One,
11    Two,
12}
13
14/// The high resolution timer. Currently, only Timer E is supported.
15pub struct HighResTimerE {
16    master: hal::stm32::HRTIM_MASTER,
17    timer: hal::stm32::HRTIM_TIME,
18    common: hal::stm32::HRTIM_COMMON,
19}
20
21impl HighResTimerE {
22    /// Construct a new high resolution timer for generating IO_update signals.
23    pub fn new(
24        timer_regs: hal::stm32::HRTIM_TIME,
25        master_regs: hal::stm32::HRTIM_MASTER,
26        common_regs: hal::stm32::HRTIM_COMMON,
27        prec: rec::Hrtim,
28    ) -> Self {
29        prec.reset().enable();
30
31        Self {
32            master: master_regs,
33            timer: timer_regs,
34            common: common_regs,
35        }
36    }
37
38    /// Configure the timer to operate in single-shot mode.
39    ///
40    /// # Note
41    /// This will configure the timer to generate a single pulse on an output channel. The timer
42    /// will only count up once and must be `trigger()`'d after / configured.
43    ///
44    /// The output will be asserted from `set_offset` to `set_offset` + `set_duration` in the count.
45    ///
46    /// # Args
47    /// * `channel` - The timer output channel to configure.
48    /// * `set_duration` - The duration that the output should be asserted for.
49    /// * `set_offset` - The first time at which the output should be asserted.
50    pub fn configure_single_shot(
51        &mut self,
52        channel: Channel,
53        delay: f32,
54        duration: f32,
55        clk: f32,
56    ) {
57        // Disable the timer before configuration.
58        self.master.mcr.modify(|_, w| w.tecen().clear_bit());
59
60        // Configure the desired timer for single shot mode with set and reset of the specified
61        // channel at the desired durations. The HRTIM is on APB2 (D2 domain), and the kernel clock
62        // is the APB bus clock.
63        let end = ((delay + duration) * clk) as u32 + 1;
64
65        // Determine the clock divider, which may be 1, 2, or 4. We will choose a clock divider that
66        // allows us the highest resolution per tick, so lower dividers are favored.
67        let div: u8 = if end < 0xFFDF {
68            1
69        } else if (end / 2) < 0xFFDF {
70            2
71        } else if (end / 4) < 0xFFDF {
72            3
73        } else {
74            panic!("Unattainable timing parameters!");
75        };
76
77        // The period register must be greater than or equal to 3 cycles.
78        let period = (end / (1 << (div - 1)) as u32) as u16;
79        assert!(period > 2);
80
81        // We now have the prescaler and the period registers. Configure the timer.
82        // Note(unsafe): The prescaler is guaranteed to be greater than or equal to 4 (minimum
83        // allowed value) due to the addition. The setting is always 1, 2, or 3, which represents
84        // all valid values.
85        self.timer
86            .timecr
87            .modify(|_, w| unsafe { w.ck_pscx().bits(div + 4) });
88
89        // Note(unsafe): The period register is guaranteed to be a 16-bit value, which will fit in
90        // this register.
91        self.timer.perer.write(|w| unsafe { w.perx().bits(period) });
92
93        // Configure the comparator 1 level.
94        let delay = (delay * clk) as u16;
95        // Note(unsafe): The offset is always a 16-bit value, so is always valid for values >= 3, as
96        // specified by the datasheet.
97        assert!(delay >= 3);
98        self.timer
99            .cmp1er
100            .write(|w| unsafe { w.cmp1x().bits(delay) });
101
102        // Configure the set/reset signals.
103        // Set on compare with CMP1, reset upon reaching PER
104        match channel {
105            Channel::One => {
106                self.timer.sete1r.write(|w| w.cmp1().set_bit());
107                self.timer.rste1r.write(|w| w.per().set_bit());
108                self.common.oenr.write(|w| w.te1oen().set_bit());
109            }
110            Channel::Two => {
111                self.timer.sete2r.write(|w| w.cmp1().set_bit());
112                self.timer.rste2r.write(|w| w.per().set_bit());
113                self.common.oenr.write(|w| w.te2oen().set_bit());
114            }
115        }
116
117        // Enable the timer now that it is configured.
118        self.master.mcr.modify(|_, w| w.tecen().set_bit());
119    }
120
121    /// Generate a single trigger of the timer to start the output pulse generation.
122    pub fn trigger(&mut self) {
123        // Generate a reset event to force the timer to start counting.
124        self.common.cr2.write(|w| w.terst().set_bit());
125    }
126}
