3/// Provide an interface for managing digital attenuators on Pounder hardware.
4///
5/// Note: The digital attenuators do not allow read-back of attenuation. To circumvent this, this
6/// driver maintains the attenuation code in both the shift register as well as the latched output
7/// register of the attenuators. This allows the "active" attenuation code to be read back by
8/// reading the shfit register. The downside of this approach is that any read is destructive, so a
9/// read-writeback approach is employed.
10pub trait AttenuatorInterface {
11    /// Set the attenuation of a single channel.
12    ///
13    /// Args:
14    /// * `channel` - The pounder channel to configure the attenuation of.
15    /// * `attenuation` - The desired attenuation of the channel in dB. This has a resolution of
16    ///   0.5dB.
17    fn set_attenuation(
18        &mut self,
19        channel: Channel,
20        attenuation: f32,
21    ) -> Result<f32, Error> {
22        let attenuation = Self::validate(attenuation)?;
23
24        // Calculate the attenuation code to program into the attenuator. The attenuator uses a
25        // code where the LSB is 0.5 dB.
26        let attenuation_code = (attenuation * 2.0) as u8;
27
28        // Read all the channels, modify the channel of interest, and write all the channels back.
29        // This ensures the staging register and the output register are always in sync.
30        let mut channels = [0_u8; 4];
31        self.transfer_attenuators(&mut channels)?;
32
33        // The lowest 2 bits of the 8-bit shift register on the attenuator are ignored. Shift the
34        // attenuator code into the upper 6 bits of the register value. Note that the attenuator
35        // treats inputs as active-low, so the code is inverted before writing.
36        channels[channel as usize] = !(attenuation_code << 2);
37        self.transfer_attenuators(&mut channels)?;
38
39        // Finally, latch the output of the updated channel to force it into an active state.
40        self.latch_attenuator(channel)?;
41
42        Ok(attenuation_code as f32 / 2.0)
43    }
44
45    /// Get the attenuation of a channel.
46    ///
47    /// Args:
48    /// * `channel` - The channel to get the attenuation of.
49    ///
50    /// Returns:
51    /// The programmed attenuation of the channel in dB.
52    fn get_attenuation(&mut self, channel: Channel) -> Result<f32, Error> {
53        let mut channels = [0_u8; 4];
54
55        // Reading the data always shifts data out of the staging registers, so we perform a
56        // duplicate write-back to ensure the staging register is always equal to the output
57        // register.
58        self.transfer_attenuators(&mut channels)?;
59        self.transfer_attenuators(&mut channels)?;
60
61        // The attenuation code is stored in the upper 6 bits of the register, where each LSB
62        // represents 0.5 dB. The attenuator stores the code as active-low, so inverting the result
63        // (before the shift) has the affect of transforming the bits of interest (and the
64        // dont-care bits) into an active-high state and then masking off the don't care bits. If
65        // the shift occurs before the inversion, the upper 2 bits (which would then be don't
66        // care) would contain erroneous data.
67        let attenuation_code = (!channels[channel as usize]) >> 2;
68
69        // Convert the desired channel code into dB of attenuation.
70        Ok(attenuation_code as f32 / 2.0)
71    }
72
73    fn reset_attenuators(&mut self) -> Result<(), Error>;
74
75    fn latch_attenuator(&mut self, channel: Channel) -> Result<(), Error>;
76
77    fn transfer_attenuators(
78        &mut self,
79        channels: &mut [u8; 4],
80    ) -> Result<(), Error>;
81
82    fn validate(attenuation: f32) -> Result<f32, Error> {
83        if !(0.0..=31.5).contains(&attenuation) {
84            Err(Error::Bounds)
85        } else {
86            Ok(attenuation)
87        }
88    }
89}
