1use super::{Channel, Error};
2
3/// Provide an interface to measure RF input power in dBm.
4pub trait PowerMeasurementInterface {
5    fn sample_converter(&mut self, channel: Channel) -> Result<f32, Error>;
6
7    /// Measure the power of an input channel in dBm.
8    ///
9    /// Args:
10    /// * `channel` - The pounder input channel to measure the power of.
11    ///
12    /// Returns:
13    /// Power in dBm after the digitally controlled attenuator before the amplifier.
14    fn measure_power(&mut self, channel: Channel) -> Result<f32, Error> {
15        let analog_measurement = self.sample_converter(channel)?;
16
17        // The AD8363 with VSET connected to VOUT provides an output voltage of 51.7 mV/dB at
18        // 100MHz with an intercept of -58 dBm.
19        // It is placed behind a 20 dB tap.
20        Ok(analog_measurement * (1. / 0.0517) + (-58. + 20.))
21    }
22}
