1use arbitrary_int::u5;
2use fugit::MegahertzU32 as MegaHertz;
3
4/// The system clock, used in various timer calculations
5pub const SYSCLK: MegaHertz = MegaHertz::MHz(400);
6
7/// The ADC setup time is the number of seconds after the CSn line goes low before the serial clock
8/// may begin. This is used for performing the internal ADC conversion.
9pub const ADC_SETUP_TIME: f32 = 220e-9;
10
11/// The maximum DAC/ADC serial clock line frequency. This is a hardware limit.
12pub const ADC_DAC_SCK_MAX: MegaHertz = MegaHertz::MHz(50);
13
14/// The optimal counting frequency of the hardware timers used for timestamping and sampling.
15pub const TIMER_FREQUENCY: MegaHertz = MegaHertz::MHz(100);
16pub const TIMER_PERIOD: f32 = 1. / (TIMER_FREQUENCY.to_Hz() as f32);
17
18/// The QSPI frequency for communicating with the pounder DDS.
19pub const POUNDER_QSPI_FREQUENCY: MegaHertz = MegaHertz::MHz(50);
20
21/// The delay after initiating a QSPI transfer before asserting the IO_Update for the pounder DDS.
22// Pending Pounder Profile writes are up to 32 bytes (QSPI FIFO depth),
23// with 2 cycles required per byte, coming out to a total of 64 QSPI clock cycles.
24// The QSPI is configured for 50MHz, so this comes out to an offset
25// of 1280 ns. We use 1300 ns to be safe.
26pub const POUNDER_IO_UPDATE_DELAY: f32 = 1_300e-9;
27
28/// The duration to assert IO_Update for the pounder DDS.
29// IO_Update should be latched for 4 SYNC_CLK cycles after the QSPI profile write. With pounder
30// SYNC_CLK running at 100MHz (1/4 of the pounder reference clock of 500MHz), this corresponds to
31// 32ns. To accomodate rounding errors, we use 50ns instead.
32pub const POUNDER_IO_UPDATE_DURATION: f32 = 50e-9;
33
34/// The DDS reference clock frequency in MHz.
35pub const DDS_REF_CLK: MegaHertz = MegaHertz::MHz(100);
36
37/// The multiplier used for the DDS reference clock PLL.
38pub const DDS_MULTIPLIER: u5 = u5::new(5);
39
40/// The DDS system clock frequency after the internal PLL multiplication.
41#[allow(dead_code)]
42pub const DDS_SYSTEM_CLK: MegaHertz =
43    MegaHertz::MHz(DDS_REF_CLK.to_MHz() * DDS_MULTIPLIER.value() as u32);
44
45/// The divider from the DDS system clock to the SYNC_CLK output (sync-clk is always 1/4 of sysclk).
46#[allow(dead_code)]
47pub const DDS_SYNC_CLK_DIV: u8 = 4;
48
49/// The maximum ADC/DAC sample processing buffer size.
50pub const MAX_SAMPLE_BUFFER_SIZE: usize = 32;
51
52pub type SampleBuffer = [u16; MAX_SAMPLE_BUFFER_SIZE];
