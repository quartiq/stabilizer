10pub const BATCH_SIZE: usize = 8;
11// 100 MHz timer, 128 divider: period of 1.28 µs, ~781.25 KHz.
12pub const SAMPLE_TICKS: u32 = 1 << 7;
13const S_PER_SAMPLE: f32 = SAMPLE_TICKS as f32 / 100e6;
14const TURN_PER_LSB: f32 = ((1u64 << 32) as f32).recip();
15const HZ_PER_LSB: f32 = ((1u64 << 32) as f32 * S_PER_SAMPLE).recip();
16
17pub const UNITS: pid::Units<f32> = pid::Units {
18    t: BATCH_SIZE as f32 * S_PER_SAMPLE,
19    x: TURN_PER_LSB,
20    y: HZ_PER_LSB,
21};

23#[derive(Debug, Clone, Default)]
24pub struct MpllState {
25    /// Lowpass state
26    lp: [DirectForm<i32, 4>; 2],
27    /// PID state
28    iir: DirectForm1<i32>,
29    /// Current modulation phase
30    phase: Wrapping<i32>,
31    /// LO samples for downconverting the next batch

33}

35impl MpllState {
36    pub fn set_frequency(&mut self, f: f32) {
37        self.iir.set_y((f * UNITS.y.recip()) as _);
38    }

122        Mpll {

124            iir: self.iir.build(&UNITS),

128        }

132impl Mpll {
133    /// Ingest and process a batch of input samples and output new modulation
134    pub fn process(
135        &self,
136        state: &mut MpllState,
137        x: [&[u16; BATCH_SIZE]; 2],
138        y: [&mut [u16; BATCH_SIZE]; 2],
139    ) -> Stream {
140        // scratch
141        let mut mix = [[0; BATCH_SIZE]; 4];
142        let [m00, m01, m10, m11] = mix.each_mut();
143        // mix
144        for (x, (mix, lo)) in x[0].iter().zip(x[1].iter()).zip(
145            m00.iter_mut()
146                .zip(m01)
147                .zip(m10.iter_mut().zip(m11))
148                .zip(state.lo[0].iter().zip(state.lo[1].iter())),
149        ) {
150            // ADC encoding, mix, 1 bit loss, 30 bit amplitude
151            *mix.0.0 = ((*x.0 as i16 as i64 * *lo.0 as i64) >> 16) as i32;
152            *mix.0.1 = ((*x.0 as i16 as i64 * *lo.1 as i64) >> 16) as i32;
153            *mix.1.0 = ((*x.1 as i16 as i64 * *lo.0 as i64) >> 16) as i32;
154            *mix.1.1 = ((*x.1 as i16 as i64 * *lo.1 as i64) >> 16) as i32;
155        }
156        // lowpass, 1 bit DC gain, 1 bit loss to 2f
157        for (mix, state) in mix.iter_mut().zip(state.lp.iter_mut()) {
158            self.lp.inplace(state, mix);
159        }
160        // decimate
161        let demod = [
162            [mix[0][BATCH_SIZE - 1], mix[1][BATCH_SIZE - 1]],
163            [mix[2][BATCH_SIZE - 1], mix[3][BATCH_SIZE - 1]],
164        ];
165        // phase
166        // need full atan2, rotation, or addtl osc to support any phase offset
167        let mut phase = Wrapping(idsp::atan2(demod[0][1], demod[0][0]));
168        // Delay correction
169        phase += Wrapping(-10) * Wrapping(state.iir.y0());
170        // PID
171        let frequency = Wrapping(self.iir.process(&mut state.iir, phase.0));
172        // modulate
173        let [y0, y1] = state.lo.each_mut();
174        let [yo0, yo1] = y;
175        for (y, yo) in y0.iter_mut().zip(y1).zip(yo0.iter_mut().zip(yo1)) {
176            state.phase += frequency;
177            (*y.0, *y.1) = idsp::cossin(state.phase.0);
178            *yo.0 = (((*y.0 as i64 * self.amplitude[0].inner as i64) >> 32)
179                as i16)
180                .wrapping_add(i16::MIN) as u16; // DAC encoding
181            *yo.1 = (((phase.0 as i64 * self.amplitude[1].inner as i64) >> 32)
182                as i16)
183                .wrapping_add(i16::MIN) as u16; // DAC encoding
184        }
185        Stream {
186            demod,
187            phase,
188            frequency,
189        }
190    }
191}
