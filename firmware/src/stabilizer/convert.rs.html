5#[derive(Debug, Serialize, Deserialize, Default)]
6#[bitenum(u2, exhaustive = true)]
7pub enum Gain {
8    #[default]
9    G1 = 0b00,
10    G2 = 0b01,
11    G5 = 0b10,
12    G10 = 0b11,
13}
14
15impl Gain {
16    /// Get the AFE gain as a numerical value.
17    pub const fn gain(self) -> f32 {
18        match self {
19            Gain::G1 => 1.0,
20            Gain::G2 => 2.0,
21            Gain::G5 => 5.0,
22            Gain::G10 => 10.0,
23        }
24    }
25}

27/// A type representing an ADC sample.
28#[derive(Copy, Clone, Default)]
29pub struct AdcCode(pub u16);
30
31impl AdcCode {
32    // The ADC has a differential input with a range of +/- 4.096 V and 16-bit resolution.
33    // The gain into the two inputs is 1/5.
34    const FULL_SCALE: f32 = 5.0 / 2.0 * 4.096;
35    const VOLT_PER_LSB: f32 = -Self::FULL_SCALE / i16::MIN as f32;
36    const LSB_PER_VOLT: f32 = 1. / Self::VOLT_PER_LSB;
37}
38
39impl From<u16> for AdcCode {
40    /// Construct an ADC code from a provided binary (ADC-formatted) code.
41    fn from(value: u16) -> Self {
42        Self(value)
43    }
44}
45
46impl From<i16> for AdcCode {
47    /// Construct an ADC code from the stabilizer-defined code (i16 full range).
48    fn from(value: i16) -> Self {
49        Self(value as u16)
50    }
51}
52
53impl From<AdcCode> for i16 {
54    /// Get a stabilizer-defined code from the ADC code.
55    fn from(code: AdcCode) -> i16 {
56        code.0 as i16
57    }
58}
59
60impl From<AdcCode> for u16 {
61    /// Get an ADC-frmatted binary value from the code.
62    fn from(code: AdcCode) -> u16 {
63        code.0
64    }
65}
66
67impl From<AdcCode> for f32 {
68    /// Convert raw ADC codes to/from voltage levels.
69    ///
70    /// # Note
71    /// This does not account for the programmable gain amplifier at the signal input.
72    fn from(code: AdcCode) -> f32 {
73        i16::from(code) as f32 * AdcCode::VOLT_PER_LSB
74    }
75}
76
77impl TryFrom<f32> for AdcCode {
78    type Error = ();
79
80    fn try_from(voltage: f32) -> Result<AdcCode, ()> {
81        let code = voltage * Self::LSB_PER_VOLT;
82        if !(i16::MIN as f32..=i16::MAX as f32).contains(&code) {
83            Err(())
84        } else {
85            Ok(AdcCode::from(code as i16))
86        }
87    }
88}

90/// Custom type for referencing DAC output codes.
91/// The internal integer is the raw code written to the DAC output register.
92#[derive(Copy, Clone, Default)]
93pub struct DacCode(pub u16);
94impl DacCode {
95    // The DAC output range in bipolar mode (including the external output op-amp) is +/- 4.096
96    // V with 16-bit resolution. The anti-aliasing filter has an additional gain of 2.5.
97    pub const FULL_SCALE: f32 = 4.096 * 2.5;
98    pub const VOLT_PER_LSB: f32 = -Self::FULL_SCALE / i16::MIN as f32;
99    pub const LSB_PER_VOLT: f32 = 1. / Self::VOLT_PER_LSB;
100}
101
102impl TryFrom<f32> for DacCode {
103    type Error = ();
104
105    fn try_from(voltage: f32) -> Result<DacCode, ()> {
106        let code = voltage * Self::LSB_PER_VOLT;
107        if !(i16::MIN as f32..=i16::MAX as f32).contains(&code) {
108            Err(())
109        } else {
110            Ok(DacCode::from(code as i16))
111        }
112    }
113}
114
115impl From<DacCode> for f32 {
116    fn from(code: DacCode) -> f32 {
117        i16::from(code) as f32 * DacCode::VOLT_PER_LSB
118    }
119}
120
121impl From<DacCode> for i16 {
122    fn from(code: DacCode) -> i16 {
123        (code.0 as i16).wrapping_sub(i16::MIN)
124    }
125}
126
127impl From<i16> for DacCode {
128    /// Encode signed 16-bit values into DAC offset binary for a bipolar output configuration.
129    fn from(value: i16) -> Self {
130        Self(value.wrapping_add(i16::MIN) as u16)
131    }
132}

134impl From<u16> for DacCode {
135    /// Create a dac code from the provided DAC output code.

138    }
