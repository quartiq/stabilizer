16pub const BATCH_SIZE: usize = 8;
17pub const SAMPLE_TICKS_E: usize = 7;
18
19// signal: p
20// demod: t'f/N
21// t': fixstate
22// -> p + t'f/N
23// -> lp(p + t'f/N) + (t - t')*f/N
24// q: pll time
25// t: true sample time
26// t = q * (PLL_M << PLL_E)
27const PLL_M: Wrapping<i32> = Wrapping(5); // 5 ns
28const PLL_E: usize =
29    16 // PLL input align ASL
30    + SAMPLE_TICKS_E // sample timer divider
31    + 1 // 10ns timer clock: 5 ns M, 1 E
32    - 1 // 2 ns DDS sample clock
33    - 2 // SYNC divider
34    - 2 // ETR prescaler
35;
36// in q32: (t - t')*PLL_M/N << 32 = (q << (32 - PLL_E - N_LOG2)) - (PLL_M*t' << (32 - N_LOG2))
37
38const _: () = {
39    assert!(0x140_0000 / BATCH_SIZE as i32 == PLL_M.0 << PLL_E);
40};
41
42const N_E: usize = 4;

44/// Fixed LO LUT mixer
45#[derive(Debug, Clone)]
46struct FixMix<const N: usize> {
47    lo: [Q32<32>; N],
48    f: usize,
49    pi_half: usize,
50}
51
52#[derive(Debug, Clone, Default)]
53struct FixMixState {
54    i: usize,
55}
56
57impl<const N: usize> FixMix<N> {
58    pub fn new(f: usize, p: f32) -> Self {
59        Self {
60            f,
61            lo: core::array::from_fn(|i| {
62                Q32::from_f64(
63                    (0.5 - 1e-5)
64                        * ((p as f64 + (i * f) as f64)
65                            * const { core::f64::consts::TAU / N as f64 })
66                        .cos(),
67                )
68            }),
69            pi_half: (0..N).find(|i| (i * f) % N == N / 4).unwrap_or_default(),
70        }
71    }
72}
73
74impl<const N: usize> SplitProcess<i32, Complex<i32>, FixMixState>
75    for FixMix<N>
76{
77    fn process(&self, state: &mut FixMixState, x: i32) -> Complex<i32> {
78        state.i = state.i.wrapping_add(1);
79        Complex([
80            x * self.lo[state.i % N],
81            x * self.lo[state.i.wrapping_sub(self.pi_half) % N],
82        ])
83    }
84}

86const WRAP_BITS: u32 = 13;

88pub const PHASE_UNITS: Units<f32> = Units {
89    // 100 MHz timer, 128 divider, BATCH_SIZE 8: period of 10.24 µs, ~98 KHz
90    t: (BATCH_SIZE << SAMPLE_TICKS_E) as f32 * 10e-9,
91    x: 1.0 / (PLL_M.0 << (32 - WRAP_BITS)) as f32,
92    // 500 MHz DDS clock
93    y: 500e6 / (1u64 << 32) as f32,
94};

96pub const AMPLITUDE_UNITS: Units<f32> = Units {
97    t: 10e-3,
98    x: 2.048 / (1 << 16) as f32, // 2.048 V per 16 bit
99    y: 1.0 / i32::MAX as f32,
100};

191impl Channel {
192    pub fn build(&self) -> ChannelConfig {
193        const { assert!(u4::MAX.value() == (1 << N_E) as u8 - 1) };
194        ChannelConfig {
195            lo: FixMix::new(self.lo_frequency.value() as _, self.lo_phase),
196            lowpass: Cascade(self.lowpass.each_ref().map(|c| Biquad::from(*c))),
197            min_power: P32::from_f32(self.min_power),
198            phase: self.phase.build(&PHASE_UNITS),
199            amplitude: self.amplitude.build(&AMPLITUDE_UNITS),
200        }
201    }

204#[derive(Debug, Clone, Default)]
205pub struct ChannelState {
206    lo: FixMixState,
207    lowpass: Complex<[DirectForm1<i32>; 3]>,
208    pub demod: Complex<i32>,
209    pub unwrap: Unwrapper<i64>,
210    pub clamp: ClampWrap<Wrapping<i32>>,
211    pub phase: DirectForm1Dither,
212    pub amplitude: DirectForm1<i32>,
213    pub hold: bool,
214    pub holds: Wrapping<u32>,
215    pub blanks: Wrapping<u32>,
216    pub slips: Wrapping<u32>,
217    pub power: P32<28>,
218}
219
220impl ChannelConfig {
221    #[inline(always)]
222    pub fn demodulate<const B: usize>(
223        &self,
224        state: &mut ChannelState,
225        x: &[u16; B],
226        y: &mut [u16; B],
227    ) {
228        // bit are peak amplitude
229        // x: 15 bit, 16 asl
230        // lo: 31 bit, 32 asr: 1 bit loss
231        let mut m =
232            x.map(|x| self.lo.process(&mut state.lo, (x as i16 as i32) << 16));
233        // m: 29 bit 0f plus 29 bit 2f
234        // lp: 1 bit DC gain
235        let [si, sq] = state.lowpass.each_mut();
236        for (c, s) in self.lowpass.0.iter().zip(si.iter_mut().zip(sq)) {
237            for m in m.iter_mut() {
238                m[0] = c.process(s.0, m[0]);
239                m[1] = c.process(s.1, m[1]);
240            }
241            // demod: 30 bit
242        }
243        state.demod = m[B - 1];
244        for (y, m) in y.iter_mut().zip(m) {
245            // output in phase (before CPU clock phase correction from PLL)
246            *y = ((m.re() >> 15) as i16).wrapping_add(i16::MIN) as u16;
247        }
248    }
249
250    #[inline(always)]
251    pub fn update(&self, state: &mut ChannelState, t: Wrapping<i32>) {
252        let d2 = state.demod.0.map(|x| x as i64 * x as i64);
253        // d2: 30 + 30 bit
254        state.power = P32::new(((d2[0] + d2[1]) >> 32) as _);
255        // power: 28 bit
256        let blank = state.power < self.min_power;
257        if blank {
258            state.blanks += Wrapping(1);
259        }
260        if state.hold {
261            state.holds += Wrapping(1);
262        }
263        if blank | state.hold {
264            state.unwrap = Default::default();
265            state.clamp = Default::default();
266            state.phase = Default::default();
267        } else {
268            let phase = state.demod.arg();
269            let phase = PLL_M * phase
270                + Wrapping(self.lo.f as i32)
271                    * ((t << (32 - PLL_E - N_E))
272                        - (PLL_M << (32 - N_E)) * Wrapping(state.lo.i as i32));
273            let dphase = Wrapping(state.unwrap.process(phase.0));
274            // |delta phase| > pi/2 indicates a possible undetected phase slip
275            // Neither a necessary nor a sufficient condition though
276            if dphase + Wrapping(1 << 30) < Wrapping(0) {
277                state.slips += Wrapping(1);
278            }
279            let phase = Wrapping((state.unwrap.y >> WRAP_BITS) as i32);
280            let phase = state.clamp.process(phase);
281            self.phase.process(&mut state.phase, phase.0);
282        }
283    }
284
285    pub fn power(&self, state: &mut ChannelState, x: u32) {
286        if state.hold {
287            Biquad::<Q32<29>>::HOLD.process(&mut state.amplitude, x as _)
288        } else {
289            self.amplitude.process(&mut state.amplitude, x as _)
290        };
291    }
