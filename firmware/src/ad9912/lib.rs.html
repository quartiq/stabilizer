76impl Serial {
77    pub fn mirror(self) -> Self {
78        let v = self.raw_value();
79        Self::new_with_raw_value(
80            v & 0x0f
81                | ((v & 1) << 7)
82                | ((v & 2) << 5)
83                | ((v & 4) << 3)
84                | ((v & 8) << 1),
85        )
86    }

139impl Pll {
140    pub fn set_refclk(&mut self, ndiv: u5, refclk: f64) -> f64 {
141        let sysclk = refclk
142            * ((self.ref_doubler() as u8 + 1) * 2 * (ndiv.value() + 2)) as f64;
143        *self = if sysclk > 900e6 {
144            self.with_vco_auto_range(false).with_vco_range_high(true)
145        } else if sysclk > 810e6 {
146            self.with_vco_auto_range(true)
147        } else {
148            self.with_vco_auto_range(false).with_vco_range_high(false)
149        };
150        sysclk
151    }
152}

173pub fn frequency_to_ftw(frequency: f64, sysclk: f64) -> u48 {
174    let lsb = sysclk.recip() * (1u64 << 48) as f64;
175    // Alias into Nyquist
176    u48::new(((frequency * lsb).round() as i64 as u64) & u48::MASK)
177}
178
179pub fn phase_to_pow(phase: f32) -> u14 {
180    // Alias into Nyquist
181    u14::new(((phase * (1u32 << 14) as f32).round() as i32 as u16) & u14::MASK)
182}

184pub fn dac_fs_to_fsc(dac_fs: f32, r_dac_ref: f32) -> u10 {
185    let lsb = r_dac_ref * (1024.0 / 192.0 / 1.2);
186    let fsc = dac_fs * lsb - (1024.0 / 192.0 * 72.0);
187    // Clamp
188    u10::new(
189        fsc.round()
190            .clamp(u10::MIN.value() as _, u10::MAX.value() as _) as _,
191    )
192}

199    fn write(&mut self, addr: Addr, data: &[u8]) -> Result<(), Error> {
200        Ok(self.bus.transaction(&mut [
201            Operation::Write(
202                &Instruction::builder()
203                    .with_addr(addr)
204                    .with_size(data.len().into())
205                    .with_read(false)
206                    .build()
207                    .raw_value()
208                    .to_be_bytes(),
209            ),
210            Operation::Write(data),
211        ])?)
212    }
213
214    fn read(&mut self, addr: Addr, data: &mut [u8]) -> Result<(), Error> {
215        Ok(self.bus.transaction(&mut [
216            Operation::Write(
217                &Instruction::builder()
218                    .with_addr(addr)
219                    .with_size(data.len().into())
220                    .with_read(true)
221                    .build()
222                    .raw_value()
223                    .to_be_bytes(),
224            ),
225            Operation::Read(data),
226        ])?)
227    }
228
229    pub fn init(&mut self) -> Result<(), Error> {
230        self.write(
231            Addr::Serial,
232            &Serial::builder()
233                .with_sdo_active(true)
234                .with_lsb_first(false)
235                .with_soft_reset(false)
236                .with_long_insn(true)
237                .build()
238                .mirror()
239                .raw_value()
240                .to_be_bytes(),
241        )?;
242        let mut id = [0; 2];
243        self.read(Addr::PartId, &mut id)?;
244        let id = u16::from_be_bytes(id);
245        if id != 0x1982 {
246            return Err(Error::Id(id));
247        }
248        Ok(())
249    }
